const BASE_API = window.location.hostname === "localhost" || window.location.hostname === "127.0.0.1" || window.location.hostname === ""

getHeaders na requisiçao const getHeaders = () => {
            const token = localStorage.getItem('auth_token');
            const headers = { 'Content-Type': 'application/json' };
            if (token) headers['Authorization'] = `Bearer ${token}`;
            return headers;
        };

  "date": "2026-02-12", definido pelo usuário

GEt BASE_API/projects?status=ACTIVE

PATCH projects BASE_API/{projectID}/status
